import React, { useState } from "react";
import MainLayout from "../../Layout/Mainlayout";
import { Link, router, usePage } from "@inertiajs/react";
import FlashMessage from "../../Component/FlashMessage.jsx";
import Select from 'react-select';

function AllTypeEmployeeReport() {
    const {  flash,departments,designations,religions,working_status } = usePage().props;
    const [employeeReport , setEmployeeReport] = useState(false);
    const [employeeReportDateRange , setEmployeeReportDateRange] = useState(false);
    const [values, setValues] = useState({
        from_date:"",
        to_date:"",
    });

    function handleSelectChange(e) {
        const selectedValue = e.target.value; // Get the selected value

        if (selectedValue === "all") {
            setEmployeeReport(false)
        } else if (selectedValue === "dept") {
            setEmployeeReport(true)
        }

    }
    function handleSelectDateChange(e) {
        const selectedValue = e.target.value; // Get the selected value

        if (selectedValue === "no") {
            setEmployeeReportDateRange(false)
        } else if (selectedValue === "yes") {
            setEmployeeReportDateRange(true)
        }

    }

    function handleChange(e) {
        const key = e.target.id;
        const value = e.target.value;
        setValues((values) => ({
            ...values,
            [key]: value,
        }));
    }


    function handleSubmit(e) {
        e.preventDefault();
        router.get("/admin/all-emp-attendance/date-range-wise-report",values);
    }
    return (
        <>
            <FlashMessage flash={flash} />
            <div className="panel flex items-center overflow-x-auto whitespace-nowrap p-3 ">
                <div className="rounded-full bg-primary p-1.5 text-white ring-2 ring-primary/30 ltr:mr-3 rtl:ml-3">
                    <svg
                        width="24"
                        height="24"
                        viewBox="0 0 24 24"
                        fill="none"
                        xmlns="http://www.w3.org/2000/svg"
                        className="h-3.5 w-3.5"
                    >
                        <path
                            d="M19.0001 9.7041V9C19.0001 5.13401 15.8661 2 12.0001 2C8.13407 2 5.00006 5.13401 5.00006 9V9.7041C5.00006 10.5491 4.74995 11.3752 4.28123 12.0783L3.13263 13.8012C2.08349 15.3749 2.88442 17.5139 4.70913 18.0116C9.48258 19.3134 14.5175 19.3134 19.291 18.0116C21.1157 17.5139 21.9166 15.3749 20.8675 13.8012L19.7189 12.0783C19.2502 11.3752 19.0001 10.5491 19.0001 9.7041Z"
                            stroke="currentColor"
                            strokeWidth="1.5"
                        />
                        <path
                            opacity="0.5"
                            d="M7.5 19C8.15503 20.7478 9.92246 22 12 22C14.0775 22 15.845 20.7478 16.5 19"
                            stroke="currentColor"
                            strokeWidth="1.5"
                            strokeLinecap="round"
                        />
                    </svg>
                </div>
                <ul className="flex space-x-2 rtl:space-x-reverse">
                    <li>
                        <Link href="#" className="text-primary hover:underline">
                            Manage Shift
                        </Link>
                    </li>
                    <li className="before:content-['/'] ltr:before:mr-2 rtl:before:ml-2">
                        <span>Add</span>
                    </li>
                </ul>
            </div>
            <div className="pt-5 grid lg:grid-cols-1 grid-cols-1 gap-6">
                <div className="panel" id="forms_grid">
                    <div className="mb-5">
                            <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
                                <div>
                                    <label>
                                        Select All Employee or Department
                                    </label>
                                    <select
                                        id="emp_dept"
                                        className="form-select text-white-dark"
                                        onChange={handleSelectChange}
                                        defaultValue="all"
                                    >
                                        <option value="all">All</option>
                                        <option value="dept">Department</option>
                                    </select>
                                </div>
                                <div>
                                    <label>
                                        Set Date Range
                                    </label>
                                    <select
                                        id="date_range"
                                        className="form-select text-white-dark"
                                        onChange={handleSelectDateChange}
                                        defaultValue="no"
                                    >
                                        <option value="yes">Yes</option>
                                        <option value="no">No</option>
                                    </select>
                                </div>
                            </div>
                    </div>
                </div>
            </div>
            {(employeeReport || employeeReportDateRange) &&
                <form
                    className="space-y-5"
                    onSubmit={handleSubmit}
                    method="post"
                >
                    <div className="pt-5 grid lg:grid-cols-1 grid-cols-1 gap-6">
                        <div className="panel" id="forms_grid">
                            <div className="mb-5">
                                <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
                                    <div>
                                        <label>
                                            Select All Employee or Department
                                        </label>
                                        <select
                                            id="emp_dept"
                                            className="form-select text-white-dark"
                                            onChange={handleSelectChange}
                                            defaultValue="all"
                                        >
                                            <option value="all">All</option>
                                            <option value="dept">Department</option>
                                        </select>
                                    </div>
                                    <div>
                                        <label>
                                            Set Date Range
                                        </label>
                                        <select
                                            id="date_range"
                                            className="form-select text-white-dark"
                                            onChange={handleSelectDateChange}
                                            defaultValue="no"
                                        >
                                            <option value="yes">Yes</option>
                                            <option value="no">No</option>
                                        </select>
                                    </div>
                                </div>

                                {employeeReport &&
                                    <div className="grid grid-cols-1 sm:grid-cols-4 gap-4">
                                        <div>
                                            <label>
                                                Select Department
                                            </label>
                                            <select
                                                id="department_id"
                                                className="form-select text-white-dark"
                                                onChange={handleSelectChange}
                                            >
                                                <option>Choose...</option>
                                                {departments.map((item) => (
                                                    <option
                                                        key={item.id}
                                                        value={item.id}
                                                    >
                                                        {item.name}
                                                    </option>
                                                ))}
                                            </select>
                                        </div>
                                        <div>
                                            <label>
                                                Select Designation
                                            </label>
                                            <select
                                                id="designation"
                                                className="form-select text-white-dark"
                                                onChange={handleSelectDateChange}
                                            >
                                                <option>Choose...</option>
                                                {designations.map((item) => (
                                                    <option
                                                        key={item.id}
                                                        value={item.id}
                                                    >
                                                        {item.name}
                                                    </option>
                                                ))}
                                            </select>
                                        </div>
                                        <div>
                                            <label>
                                                Select Religion
                                            </label>
                                            <select
                                                id="regligion_id"
                                                className="form-select text-white-dark"
                                                onChange={handleSelectDateChange}
                                            >
                                                <option>Choose...</option>
                                                {religions.map((item) => (
                                                    <option
                                                        key={item.id}
                                                        value={item.id}
                                                    >
                                                        {item.name}
                                                    </option>
                                                ))}
                                            </select>
                                        </div>
                                        <div>
                                            <label>
                                                Select Gender
                                            </label>
                                            <select
                                                id="gender"
                                                className="form-select text-white-dark"
                                                onChange={handleSelectDateChange}
                                            >
                                                <option>Choose...</option>
                                                <option value="Male">Male</option>
                                                <option value="Female">Female</option>
                                            </select>
                                        </div>
                                    </div>
                                }

                                {employeeReportDateRange &&
                                    <div className="grid grid-cols-1 sm:grid-cols-4 gap-4">
                                        <div>
                                            <label>
                                                Select Working Status
                                            </label>
                                            <select
                                                id="working_status"
                                                className="form-select text-white-dark"
                                                onChange={handleSelectDateChange}
                                            >
                                                <option>Choose...</option>
                                                {working_status.map((item) => (
                                                    <option
                                                        key={item.id}
                                                        value={item.id}
                                                    >
                                                        {item.name}
                                                    </option>
                                                ))}
                                            </select>
                                        </div>
                                        <div>
                                            <label>
                                                From Date
                                            </label>
                                            <input
                                                id="from_date"
                                                type="date"
                                                placeholder="Enter Currency"
                                                className="form-input"
                                                value={values.currency || ''}
                                                onChange={handleChange}
                                            />
                                        </div>
                                        <div>
                                            <label>
                                                To Date
                                            </label>
                                            <input
                                                id="to_date"
                                                type="date"
                                                placeholder="Enter Currency"
                                                className="form-input"
                                                value={values.currency || ''}
                                                onChange={handleChange}
                                            />
                                        </div>
                                    </div>
                                }
                            </div>

                        </div>

                    </div>
                </form>
            }
        </>
    );
}

AllTypeEmployeeReport.layout = (page) => (
    <MainLayout children={page} title="HR || Add Shift" />
);

export default AllTypeEmployeeReport;
